299 pure subroutine split_path(filepath, head, tail)
300 character(*), intent(in) :: filepath
301 character(:), allocatable, intent(out) :: head
302 character(:), allocatable, intent(out) :: tail
303 !private
304 character(:), allocatable :: temp
305 integer :: i, ipoint, isep
306
307 ! Empty string, return (.,'')
308 if (len_trim(filepath) == 0) then
309 head = '.'; tail = ''
310 return
311 end if
312
313 ! Remove trailing path separators
314 temp = trim(adjustl(filepath))
315 if (temp(len(temp):len(temp)) == separator) then
316 temp = trim(temp(:len(temp)-1))
317 else
318 ipoint = index(filepath, '.', back=.true.)
319 isep = index(filepath, separator, back=.true.)
320 if (ipoint > isep .and. isep > 0) then
321 temp = trim(temp(:isep-1))
322 end if
323 end if
324
325 if (len_trim(temp) == 0) then
326 head = separator
327 tail = ''
328 return
329 end if
330
331 i = len(temp) - index(temp, separator, back=.true.) + 1
332
333 ! if no `pathsep`, then it probably was a root dir like `C:\`
334 if (i == 0) then
335 head = temp // separator
336 tail = ''
337 return
338 end if
339
340 head = temp(:len(temp)-i)
341
342 ! child of a root directory
343 if (index(temp, separator, back=.true.) == 0) then
344 head = head // separator
345 end if
346
347 tail = temp(len(temp)-i+2:)
348 end subroutine
