2!! @defgroup group_parser Parser
3!! Fortran Preprocessor (fpx) â€“ core parsing and preprocessing module
4!!
5!! This module implements a full-featured, modern Fortran preprocessor supporting:
6!! - C-style line continuations with `\` and `\\`
7!! - Fortran-style `&` continuations
8!! - `#define`, `#undef`, object-like and function-like macros with variadic support
9!! - `#include` with proper path resolution and recursion guard
10!! - Conditional compilation: `#if`, `#ifdef`, `#ifndef`, `#elif`, `#else`, `#endif`
11!! - C-style `/* ... */` comments (nestable aware)
12!! - Macro expansion with argument substitution and stringification (`#`) / token-pasting (`##`)
13!! - Interactive REPL mode when reading from stdin
14!! - Multiple entry points for file-to-file, unit-to-unit, etc.
15!!
16!! The preprocessor is designed to be standards-conforming where possible while adding
17!! useful extensions (variadic macros, better diagnostics, include path handling).
18!!
19!! <h2 class="groupheader">Examples</h2>
20!!
21!! 1. Preprocess a file to stdout:
22!! @code{.f90}
23!! call preprocess('input.F90')
24!! @endcode
25!!
26!! 2. Preprocess a file and write to another file:
27!! @code{.f90}
28!! call preprocess('src/main.F90', 'preprocessed/main.F90')
29!! @endcode
30!!
31!! 3. Use in a build system with unit numbers:
32!! @code{.f90}
33!! integer :: iu, ou
34!! open(newunit=iu, file='input.F90')
35!! open(newunit=ou, file='output.F90')
36!! call preprocess(iu, ou)
37!! close(iu); close(ou)
38!! ...
39!! @endcode
40!!
41!! 4. Interactive mode (stdin to stdout):
42!! @code
43!! $ ./fpx
44!! [in] #define PI 3.1415926535
45!! [out]
46!! [in] real :: x = PI*2
47!! [out] real :: x = 3.1415926535*2
48!! [in] (empty line or 'quit' to exit)
49!! @endcode

224 !> Worker routine that reads lines, handles continuations, comments and directives
225 !! This is the main loop that:
226 !! - reads lines with interactive prompt when iunit==stdin
227 !! - handles both `\` and `&` continuations
228 !! - strips or preserves comments appropriately
229 !! - calls process_line() for directive processing and macro expansion
230 !! - stitches lines when Fortran continuation (`&`) is active
231 !! @param[in] iunit Input unit
232 !! @param[in] ounit Output unit
233 !! @param[inout] macros(:) Current macro table (passed by value between include levels)
234 !! @param[in] from_include True if called recursively from #include
235 !!
236 !! @b Remarks
237 !! @ingroup group_parser
238 subroutine preprocess_unit(iunit, ounit, macros, from_include)
239 integer, intent(in) :: iunit
240 integer, intent(in) :: ounit
241 type(macro), allocatable, intent(inout) :: macros(:)
242 logical, intent(in) :: from_include
243 !private
244 integer :: ierr, n
245 character(:), allocatable :: uline
246 logical :: interactive
247
248 interactive = iunit == stdin
249
250 if (interactive) then
251 write(*, *)
252 write(*, *) ' Welcome to fpx, the extended Fortran preprocessor. '
253 write(*, *) ' The program can be exited at any time by hitting'
254 write(*, *) " 'Enter' at the prompt without entering any data, "
255 write(*, *) " or with the 'quit' command."
256 end if
257 do
258 if (interactive) write(*, '(/a)', advance='no') ' [in] ' ! Command line prompt
259 read(iunit, '(A)', iostat=ierr) line
260
261 if (interactive) then
262 if (line == '') exit
263 uline = uppercase(trim(adjustl(line)))
264 if (uline == 'QUIT') exit
265 end if
266 if (ierr /= 0) then
267 if (ierr == iostat_end .and. from_include) f_continue = tail(tmp) == '&'
268 exit
269 end if
270 if (.not. from_include) iline = iline + 1
271
272 if (c_continue) then
273 continued_line = continued_line(:icontinuation) // trim(adjustl(line))
274 else
275 continued_line = trim(adjustl(line))
276 end if
277 n = len_trim(continued_line); if (n == 0) cycle
278
279 ! Check for line continuation with '\'
280 if (verify(continued_line(n:n), '\') == 0) then
281 ! Check for line break with '\\'
282 if (continued_line(len_trim(continued_line) - 1:len_trim(continued_line)) == '\\') then
283 c_continue = .true.
284 continued_line = continued_line(:len_trim(continued_line) - 2) // new_line('A') ! Strip '\\'
285 icontinuation = len_trim(continued_line)
286 else
287 c_continue = .true.
288 icontinuation = len_trim(continued_line) - 1
289 continued_line = continued_line(:icontinuation)
290 end if
291 cycle
292 else
293 c_continue = .false.
294
295 tmp = process_line(continued_line, ounit, name, iline, macros, stitch)
296 if (len_trim(tmp) == 0) cycle
297
298 in_comment = head(tmp) == '!'
299
300 if (merge(head(res) == '!', in_comment, len_trim(res) > 0)) then
301 f_continue = tail(tmp) == '&'
302 else
303 if (in_comment .and. f_continue) cycle
304 f_continue = .not. in_comment .and. tail(tmp) == '&'
305 end if
306
307 if (f_continue .or. stitch) then
308 reprocess = .true.
309 res = concat(res, tmp)
310 else
311 if (reprocess) then
312 if (.not. in_comment .and. head(res) == '!') then
313 write(ounit, '(A)') res
314 res = process_line(tmp, ounit, name, iline, macros, stitch)
315 else
316 res = process_line(concat(res, tmp), ounit, name, iline, macros, stitch)
317 end if
318 reprocess = .false.
319 else
320 res = trim(tmp)
321 end if
322 if (interactive) write(*, '(/a)', advance='no') ' [out] ' ! Command line prompt
323 write(ounit, '(A)') res
324 res = ''
325 end if
326 end if
327 end do
328
329 if (cond_depth > 0) then
330 if (verbose) print *, "Error: Unclosed conditional block at end of file ", trim(name)
331 end if
332 end subroutine

334 !> Process a single (possibly continued) line â€“ handles directives and macro expansion
335 !! Responsibilities:
336 !! - Strip or terminate C-style block comments (`/* ... */`)
337 !! - Detect and delegate preprocessor directives (`#define`, `#include`, conditionals, etc.)
338 !! - Perform macro expansion when the line is in an active conditional block
339 !! - Return whether the next line should be stitched (for Fortran `&` continuation inside macros)
340 !! @param[in] current_line Input line (already continued and trimmed)
341 !! @param[in] ounit Output unit (used only for diagnostics inside called routines)
342 !! @param[in] filepath Current file name (for error messages)
343 !! @param[in] linenum Current line number (for error messages)
344 !! @param[inout] macros(:) Macro table
345 !! @param[out] stch Set to .true. if the expanded line ends with `&` (stitch next line)
346 !! @return Processed line (directives removed, macros expanded)
347 !!
348 !! @b Remarks
349 !! @ingroup group_parser
350 recursive function process_line(current_line, ounit, filepath, linenum, macros, stch) result(rst)
351 character(*), intent(in) :: current_line
352 integer, intent(in) :: ounit
353 character(*), intent(in) :: filepath
354 integer, intent(in) :: linenum
355 type(macro), allocatable, intent(inout) :: macros(:)
356 logical, intent(out) :: stch
357 character(:), allocatable :: rst
358 !private
359 character(:), allocatable :: trimmed_line
360 logical :: active
361 logical, save :: l_in_comment = .false.
362 integer :: idx, comment_start, comment_end, n
363
364 trimmed_line = trim(adjustl(current_line))
365 rst = ''
366 comment_end = index(trimmed_line, '*/')
367 if (l_in_comment .and. comment_end > 0) then
368 trimmed_line = trimmed_line(comment_end + 2:)
369 l_in_comment = .false.
370 end if
371
372 if (l_in_comment) return
373 comment_start = index(trimmed_line, '/*')
374 if (comment_start > 0) then
375 trimmed_line = trimmed_line(:comment_start - 1)
376 l_in_comment = comment_end == 0
377 end if
378 n = len(trimmed_line); if (n == 0) return
379
380 active = is_active()
381 if (verbose) print *, "Processing line ", linenum, ": '", trim(trimmed_line), "'"
382 if (verbose) print *, "is_active() = ", active, ", cond_depth = ", cond_depth
383 if (head(trimmed_line) == '#') then
384 if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'DEFINE') .and. active) then
385 call handle_define(trimmed_line, macros, 'DEFINE')
386 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'UNDEF') .and. active) then
387 call handle_undef(trimmed_line, macros, 'UNDEF')
388 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'INCLUDE') .and. active) then
389 call handle_include(trimmed_line, ounit, filepath, linenum, preprocess_unit, macros, 'INCLUDE')
390 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'IFDEF')) then
391 call handle_ifdef(trimmed_line, filepath, linenum, macros, 'IFDEF')
392 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'IFNDEF')) then
393 call handle_ifndef(trimmed_line, filepath, linenum, macros, 'IFNDEF')
394 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'IF')) then
395 call handle_if(trimmed_line, filepath, linenum, macros, 'IF')
396 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'ELIF')) then
397 call handle_elif(trimmed_line, filepath, linenum, macros, 'ELIF')
398 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'ELSE')) then
399 call handle_else(filepath, linenum)
400 else if (starts_with(uppercase(adjustl(trimmed_line(2:))), 'ENDIF')) then
401 call handle_endif(filepath, linenum)
402 end if
403 else if (active) then
404 if (.not. global%expand_macros) then
405 rst = trimmed_line
406 else
407 rst = adjustl(expand_all(trimmed_line, macros, filepath, linenum, stch))
408 if (verbose) print *, "Writing to output: '", trim(rst), "'"
409 end if
410
411 end if
412 end function
