1!> @file
2!! @defgroup group_include Include
3!! Include file handling and resolution for the fpx Fortran preprocessor
4!!
5!! This module implements robust and standard-compliant processing of `#include` directives
6!! with full support for:
7!! - Both forms: `#include "file.h"` (local/user) and `#include <file.h>` (system)
8!! - Relative paths resolved against the directory of the parent source file
9!! - Search in user-defined include directories (`global%includedir`)
10!! - Fallback to current working directory
11!! - Proper error reporting with file name and line number context
12!! - Recursion safety through integration with the main preprocessor loop
13!! - Seamless integration via the abstract `preprocess` procedure pointer
14!!
15!! The routine correctly strips quotes or angle brackets, performs path resolution,
16!! checks file existence, opens the file, and recursively invokes the main preprocessing
17!! engine on the included content using the same macro environment.
18!!
19!! <h2 class="groupheader">Examples</h2>
20!!
21!! 1. Include a local header from the same directory using quote or angle brackets:
22!! @code{.f90}
23!! #include <config.h>
24!! !> fpx will look for ./config.h relative to the current source file
25!! @endcode
26!!
27!! 2. Using from the driver program (adding include paths):
28!! @code{.f90}
29!! global%includedir = ['/usr/include', './include', './headers']
30!! call preprocess('main.F90', 'main.f90')
31!! !> All #include <...> will search these directories in order
32!! @endcode
33!!
34!! 3. Verbose error reporting when a file is not found:
35!! @code{.txt}
36!! $ fpx -v src/utils.F90
37!! Error: Cannot find include file 'missing.h' at src/utils.F90:27
38!! @endcode

70 !> Process a #include directive encountered during preprocessing
71 !! Resolves the include file name (quoted or angle-bracketed), searches for the file
72 !! using standard rules (parent directory first, then global include paths, then cwd),
73 !! opens it, and recursively preprocesses its contents into the output unit.
74 !! @param[in] input Full line containing the #include directive
75 !! @param[in] ounit Output unit where preprocessed content is written
76 !! @param[in] parent_file Path of the file containing the #include

78 !! @param[in] preprocess Procedure pointer to the main line-by-line preprocessor
79 !! @param[inout] macros Current macro table (shared across recursion levels)

98 dir = dirpath(parent_file)
99 pos = index(uppercase(input), token) + len(token)
100 include_file = trim(adjustl(input(pos:)))
101 if (include_file(1:1) == '"') then
102 include_file = include_file(2:index(include_file(2:), '"'))
103 else if (include_file(1:1) == '<') then
104 include_file = include_file(2:index(include_file(2:), '>'))
105 end if

108 if (is_rooted(ifile)) then

111 include_file = ifile
112 else
113 if (verbose) print *, "Error: Cannot find include file '", trim(include_file), "' at ", trim(parent_file), ":", iline
114 return
115 end if
116 else
117 ifile = join(dir, include_file)
118 inquire(file=ifile, exist=exists)
119 if (exists) then
120 include_file = ifile
121 else
122 do i = 1, size(global%includedir)
123 ifile = join(global%includedir(i), include_file)
124 inquire(file=ifile, exist=exists)
125 if (exists) then
126 include_file = ifile
127 exit
128 end if
129 end do
130

132 ifile = join(cwd(), include_file)
133 inquire(file=ifile, exist=exists)
134 if (exists) then
135 include_file = ifile
136 else
137 if (verbose) print *, "Error: Cannot find include file '", trim(include_file), "' at ", trim(parent_file), ":", iline
138 return
139 end if
140 end if
141 end if
142 end if

145 if (ierr /= 0) then
146 if (verbose) print *, "Error: Cannot open include file '", trim(include_file), "' at ", trim(parent_file), ":", iline
147 return
148 end if

150 call preprocess(iunit, ounit, macros, .true.)
151 close (iunit)
152 end subroutine
